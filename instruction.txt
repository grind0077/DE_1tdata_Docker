1. нужно создать dockerfile, в котором необходимо описать инструкции для развертывания СУБД Postgres
   и добавить в этот образ автоматический запуск скрипта от вашего коллеги-джуна. 
   На основе данного docker-файла необходимо запустить контейнер и убедиться,
   что база действительно поднялась и скрипт вашего коллеги отработал.
2. необходимо написать специальную команду docker, которая позволит подключаться к работающему контейнеру, запускать интерфейс psql и вносить новые данные «на лету».
3. необходимо добавить в docker-файл специальную инструкцию, которая вызывает механизм для хранения данных в контейнере и убедиться, что он исправно работает.


#создать dockerfile, в котором необходимо описать инструкции для развертывания СУБД Postgres
FROM postgres:lastest
ENV POSTGRES_DB=database
ENV POSTGRES_USER=user
ENV POSTGRES_PASSWORD=password

#добавить в этот образ автоматический запуск скрипта от вашего коллеги-джуна
COPY init.sql /docker-entrypoint-initdb.b/init.sql

#На основе данного docker-файла необходимо запустить контейнер и убедиться, что база действительно поднялась и скрипт вашего коллеги отработал
Собераем Docker-контейнер:
docker build -t sql_test:latest .

#запускаем:
docker run -d -p 5432:5432 --name sql_test_container sql_test:latest


# чтобы сохранить данные между перезапусками контейнера добавим в dockerfile:
VOLUME data:/var/lib/postgresql/data

#Чтобы убедиться, что контейнер успешно запущен, выполните:
docker ps
